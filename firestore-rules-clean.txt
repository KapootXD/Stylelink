rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /users/{userId} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    match /outfits/{outfitId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}
